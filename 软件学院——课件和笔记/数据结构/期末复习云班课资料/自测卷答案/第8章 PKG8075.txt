typedef struct
{  int key;
   float info;
}ElemType;

typedef struct {
          ElemType   *elem; 
          int          length;   
}SSTable;

int binsrch(SSTable st,int k)
{  int low,high,mid,found;
   low=1;  high=st.length; found=0;
   while((low<=high)&&(found==0))
   {  mid=(low+high)/2;
      if(k>st.elem[mid].key)  low=mid+1;
      else if(k==st.elem[mid].key)  found=1;
      else   high=mid-1;
   }
   if(found==1)
      return(mid);
   else
      return(0);
}
